#script para deixar os papeis de parede random
#!/bin/bash
while true; do
    feh --randomize --bg-fill ~/Downloads/scifi/*
    sleep 600
done

sh .random_wallpaper.sh &
nano .i3/config
exec_always ~/.random_wallpaper.sh
